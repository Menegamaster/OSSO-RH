// ============================================================ // OSSO RH — Google Apps Script (Backend / API) // ============================================================ function doGet(e) { var callback = e.parameter.callback; var result; try { // POST via JSONP (payload no GET) if (e.parameter.method === 'POST' && e.parameter.payload) { var data = JSON.parse(decodeURIComponent(e.parameter.payload)); result = handlePost(data); } else { // GET normal var action = e.parameter.action; if (action === 'getFuncionarios') result = getFuncionarios(); else if (action === 'getPresenca') result = getPresenca(e.parameter.data); else if (action === 'getComissao') result = getComissao(e.parameter.data); else if (action === 'getPresencaRange') result = getPresencaRange(e.parameter.inicio, e.parameter.fim); else if (action === 'getComissaoRange') result = getComissaoRange(e.parameter.inicio, e.parameter.fim); else result = { error: 'Ação desconhecida' }; } } catch (err) { result = { error: err.message }; } var json = JSON.stringify(result); var out = callback ? ContentService.createTextOutput(callback + '(' + json + ')').setMimeType(ContentService.MimeType.JAVASCRIPT) : ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); return out; } function doPost(e) { var result; try { var data = JSON.parse(e.postData.contents); result = handlePost(data); } catch (err) { result = { error: err.message }; } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function handlePost(data) { var action = data.action; if (action === 'salvarFuncionario') return salvarFuncionario(data); if (action === 'editarFuncionario') return editarFuncionario(data); if (action === 'salvarPresenca') return salvarPresenca(data); if (action === 'salvarComissao') return salvarComissao(data); if (action === 'deletarFuncionario') return deletarFuncionario(data); return { error: 'Ação desconhecida: ' + action }; } // ============================================================ // SHEETS // ============================================================ function getSheet(nome) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(nome); if (!sheet) { sheet = ss.insertSheet(nome); if (nome === 'Funcionarios') { sheet.appendRow(['id','nome','cargo','setor','ativo', 'seg_entrada','seg_saida','seg_intervalo', 'ter_entrada','ter_saida','ter_intervalo', 'qua_entrada','qua_saida','qua_intervalo', 'qui_entrada','qui_saida','qui_intervalo', 'sex_entrada','sex_saida','sex_intervalo', 'sab_dia_entrada','sab_dia_saida','sab_dia_intervalo', 'sab_noite_entrada','sab_noite_saida','sab_noite_intervalo', 'dom_dia_entrada','dom_dia_saida','dom_dia_intervalo', 'dom_noite_entrada','dom_noite_saida','dom_noite_intervalo']); } else if (nome === 'Presenca') { sheet.appendRow(['data','funcionario_id','turno','status','horas_normais','horas_noturnas']); } else if (nome === 'Comissao') { sheet.appendRow(['data','valor_bruto','desconto_pct','valor_liquido','presentes_salao','presentes_cozinha','distribuicao_json']); } } return sheet; } // ============================================================ // FUNCIONÁRIOS // ============================================================ function getFuncionarios() { var sheet = getSheet('Funcionarios'); var data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; var headers = data[0]; var result = []; for (var i = 1; i < data.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) obj[headers[j]] = data[i][j]; if (obj.ativo !== false && obj.ativo !== 'false') result.push(obj); } return result; } function salvarFuncionario(data) { var sheet = getSheet('Funcionarios'); var id = new Date().getTime().toString(); sheet.appendRow([ id, data.nome, data.cargo, data.setor, true, data.seg_entrada||'', data.seg_saida||'', data.seg_intervalo||0, data.ter_entrada||'', data.ter_saida||'', data.ter_intervalo||0, data.qua_entrada||'', data.qua_saida||'', data.qua_intervalo||0, data.qui_entrada||'', data.qui_saida||'', data.qui_intervalo||0, data.sex_entrada||'', data.sex_saida||'', data.sex_intervalo||0, data.sab_dia_entrada||'', data.sab_dia_saida||'', data.sab_dia_intervalo||0, data.sab_noite_entrada||'', data.sab_noite_saida||'', data.sab_noite_intervalo||0, data.dom_dia_entrada||'', data.dom_dia_saida||'', data.dom_dia_intervalo||0, data.dom_noite_entrada||'', data.dom_noite_saida||'', data.dom_noite_intervalo||0 ]); return { ok: true, id: id }; } function editarFuncionario(data) { var sheet = getSheet('Funcionarios'); var rows = sheet.getDataRange().getValues(); var headers = rows[0]; var idIdx = headers.indexOf('id'); for (var i = 1; i < rows.length; i++) { if (rows[i][idIdx].toString() === data.id.toString()) { var keys = Object.keys(data); for (var k = 0; k < keys.length; k++) { var colIdx = headers.indexOf(keys[k]); if (colIdx >= 0) sheet.getRange(i+1, colIdx+1).setValue(data[keys[k]]); } return { ok: true }; } } return { error: 'Funcionário não encontrado' }; } function deletarFuncionario(data) { var sheet = getSheet('Funcionarios'); var rows = sheet.getDataRange().getValues(); var headers = rows[0]; var idIdx = headers.indexOf('id'); var aIdx = headers.indexOf('ativo'); for (var i = 1; i < rows.length; i++) { if (rows[i][idIdx].toString() === data.id.toString()) { sheet.getRange(i+1, aIdx+1).setValue(false); return { ok: true }; } } return { error: 'Não encontrado' }; } // ============================================================ // PRESENÇA // ============================================================ function salvarPresenca(data) { var sheet = getSheet('Presenca'); var rows = sheet.getDataRange().getValues(); var toDelete = []; for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][0] === data.data && rows[i][1].toString() === data.funcionario_id.toString() && rows[i][2] === data.turno) { toDelete.push(i + 1); } } for (var d = 0; d < toDelete.length; d++) sheet.deleteRow(toDelete[d] - d); sheet.appendRow([data.data, data.funcionario_id, data.turno, data.status, data.horas_normais||0, data.horas_noturnas||0]); return { ok: true }; } function getPresenca(data) { var sheet = getSheet('Presenca'); var rows = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0] === data) { result.push({ data: rows[i][0], funcionario_id: rows[i][1].toString(), turno: rows[i][2], status: rows[i][3], horas_normais: rows[i][4], horas_noturnas: rows[i][5] }); } } return result; } function getPresencaRange(inicio, fim) { var sheet = getSheet('Presenca'); var rows = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0] >= inicio && rows[i][0] <= fim) { result.push({ data: rows[i][0], funcionario_id: rows[i][1].toString(), turno: rows[i][2], status: rows[i][3], horas_normais: rows[i][4], horas_noturnas: rows[i][5] }); } } return result; } // ============================================================ // COMISSÃO // ============================================================ function salvarComissao(data) { var sheet = getSheet('Comissao'); var rows = sheet.getDataRange().getValues(); for (var i = rows.length - 1; i >= 1; i--) { if (rows[i][0] === data.data) { sheet.deleteRow(i+1); break; } } sheet.appendRow([ data.data, data.valor_bruto, data.desconto_pct, data.valor_liquido, JSON.stringify(data.presentes_salao), JSON.stringify(data.presentes_cozinha), JSON.stringify(data.distribuicao) ]); return { ok: true }; } function getComissao(data) { var sheet = getSheet('Comissao'); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] === data) { return { data: rows[i][0], valor_bruto: rows[i][1], desconto_pct: rows[i][2], valor_liquido: rows[i][3], presentes_salao: JSON.parse(rows[i][4]||'[]'), presentes_cozinha: JSON.parse(rows[i][5]||'[]'), distribuicao: JSON.parse(rows[i][6]||'{}') }; } } return null; } function getComissaoRange(inicio, fim) { var sheet = getSheet('Comissao'); var rows = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0] >= inicio && rows[i][0] <= fim) { result.push({ data: rows[i][0], valor_bruto: rows[i][1], desconto_pct: rows[i][2], valor_liquido: rows[i][3], presentes_salao: JSON.parse(rows[i][4]||'[]'), presentes_cozinha: JSON.parse(rows[i][5]||'[]'), distribuicao: JSON.parse(rows[i][6]||'{}') }); } } return result; } // ============================================================ // SEED // ============================================================ function seedFuncionarios() { var sheet = getSheet('Funcionarios');
